async function handleSend() { const input = document.getElementById('user-input'); const box = document.getElementById('chat-box'); const message = input.value; if(!message) return; // 1. إضافة رسالتك فوراً box.innerHTML += `
أنت: ${message}
`; input.value = ''; box.scrollTop = box.scrollHeight; // 2. إظهار حالة "جاري التفكير" const loaderId = 'loader-' + Date.now(); box.innerHTML += `
Pigrok يحلل...
`; try { // 3. ربط المحرك (هذا هو الجزء الذي يحول النظام من "جامد" إلى "ذكي") // ملاحظة: لكي يعمل هذا، يجب أن تضع رابط الـ Backend الخاص بك هنا const response = await fetch('https://api.your-ai-engine.com/v1/chat', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ prompt: message }) }); const result = await response.json(); // 4. استبدال "جاري التحليل" برد الذكاء الاصطناعي الحقيقي document.getElementById(loaderId).innerHTML = `Pigrok: ${result.reply}`; } catch (e) { document.getElementById(loaderId).innerHTML = `خطأ في الاتصال بالمحرك. يرجى التأكد من السيرفر.`; } box.scrollTop = box.scrollHeight; }
